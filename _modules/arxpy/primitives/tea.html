[docs]    @classmethod
    def eval(cls, x, y):
        v0 = x
        v1 = y
        s = Constant(0, 32)
        delta = Constant(0x9e3779b9, 32)
        k0, k1, k2, k3 = cls.round_keys
        for i in range(cls.rounds):
            s += delta
            v0 += ((v1 << Constant(4, 32)) + k0) ^ (v1 + s) ^ ((v1 >> Constant(5, 32)) + k1)
            v1 += ((v0 << Constant(4, 32)) + k2) ^ (v0 + s) ^ ((v0 >> Constant(5, 32)) + k3)
        return v0, v1
